## Install
```shell
npm install monkey-ui -S
```

## Quick Start
``` javascript
import Vue from 'vue'
import MonkeyUI from 'monkey-ui'

Vue.use(MonkeyUI)

// or
import {
  Select,
  Button
  // ...
} from 'monkey-ui'

Vue.component(Select.name, Select)
Vue.component(Button.name, Button)
```
