## Install
```shell
npm install monkey-view -S
```

## Quick Start
``` javascript
import Vue from 'vue'
import MonkeyView from 'monkey-view'

Vue.use(MonkeyView)

// or
import {
  Select,
  Button
  // ...
} from 'monkey-view'

Vue.component(Select.name, Select)
Vue.component(Button.name, Button)
```
